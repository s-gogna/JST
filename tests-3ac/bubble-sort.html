Bubble Sort

Input Source 

const int N_ITEMS = 5;
int bubble_sort(int items[N_ITEMS], int n_items);

int main() {
  int items[N_ITEMS] = {5, 1, 4, 3, 2};

  bubble_sort(items, 5);

  return 0;
}

int bubble_sort(int items[N_ITEMS], int n_items) {
  int i, j;
  int temp;

  for (i = 0; i < n_items; i++) {
    for (j = i; j < n_items; j++) {
      if (items[i] < items[j]) {
        temp = items[i];
        items[i] = items[j];
        items[j] = temp;
      }
    }
  }
}
